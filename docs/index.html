# return true if there is a positive association between age and FEV1 values of children
# assume linear association
isPostivelyAssociated <- function(dataset) {
  corrcoef <- cor(dataset$age, dataset$fev1, use = "complete.obs")
  return(corrcoef > 0)
}
isPostivelyAssociated(dataset)

plot_scatter <- function(dataset) {
  
  # --- Simulate positively correlated data ---

    theme_minimal()

## Function for supporting premise statement 2
isSlopeCI <- function(dataset) {
  # Fit linear model
  model <- lm(fev1 ~ age,, data = dataset)
  
  # Get 95% confidence interval for slope (X coefficient)
  ci <- confint(model, level = 0.95)["age", ]  # Extract slope CI
  
  # Check conditions
  return((ci[1] > 0) && (ci[2] > 0))
}

isSlopeCI(dataset)
## [1] TRUE

Since the function returns TRUE, we know that the 95% confidence interval for the regression slope is positive.
